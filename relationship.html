-- 1. Create the Projects table
CREATE TABLE Projects (
ProjectID INT IDENTITY(1,1) PRIMARY KEY,
ProjectName NVARCHAR(50)
);
-- 2. Create the Assignments table
CREATE TABLE Assignments (
AssignmentID INT IDENTITY(1,1) PRIMARY KEY,
AssignmentName NVARCHAR(50),
ProjectID INT
);
-- 3. Add the Foreign Key relationship
ALTER TABLE Assignments
ADD CONSTRAINT FK_Assignments_Projects
FOREIGN KEY (ProjectID)
REFERENCES Projects(ProjectID)
ON DELETE CASCADE; -- Optional: Deletes assignments if the parent project is deleted
