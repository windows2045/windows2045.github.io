/* =========================================================
SCRIPT 2 - ASSIGN ROLE TO USER BY EMAIL
Database: EnvMgmt2026
========================================================= */
USE EnvMgmt2026;
GO
/* =========================================
VARIABLES
========================================= */
--DECLARE @Email NVARCHAR(256) = 'victor.azula@lacity.org';
--DECLARE @RoleName NVARCHAR(256) = 'SysAdmin';
DECLARE @Email NVARCHAR(256) = 'lyndsay.nash@lacity.org';
DECLARE @RoleName NVARCHAR(256) = 'Admin';
-- DECLARE @RoleName NVARCHAR(256) = 'Administrator';
/* =========================================
GET USER ID
========================================= */
DECLARE @UserId NVARCHAR(450);
SELECT @UserId = Id
FROM AspNetUsers
WHERE NormalizedEmail = UPPER(@Email);
/* =========================================
GET ROLE ID
========================================= */
DECLARE @RoleId NVARCHAR(450);
SELECT @RoleId = Id
FROM AspNetRoles
WHERE NormalizedName = UPPER(@RoleName);
/* =========================================
VALIDATION
========================================= */
IF @UserId IS NULL
BEGIN
PRINT 'ERROR: User email not found.';
RETURN;
END
IF @RoleId IS NULL
BEGIN
PRINT 'ERROR: Role not found.';
RETURN;
END
/* =========================================
ASSIGN ROLE IF NOT ALREADY ASSIGNED
========================================= */
IF NOT EXISTS
(
SELECT 1
FROM AspNetUserRoles
WHERE UserId = @UserId
AND RoleId = @RoleId
)
BEGIN
INSERT INTO AspNetUserRoles
(
UserId,
RoleId
)
VALUES
(
@UserId,
@RoleId
);
PRINT 'Role assigned successfully.';
END
ELSE
BEGIN
PRINT 'User already has this role.';
END
/* =========================================
VERIF
y ========================================= */
SELECT
U.Email,
R.Name AS RoleName
FROM AspNetUsers U
INNER JOIN AspNetUserRoles UR
ON U.Id = UR.UserId
INNER JOIN AspNetRoles R
ON UR.RoleId = R.Id
WHERE U.Email = @Email;
GO
