/* =========================================================
SCRIPT 1 - CREATE ASP.NET CORE IDENTITY ROLES
Database: EnvMgmt2026
========================================================= */
USE EnvMgmt2026;
GO
/* Create roles only if they do not already exist */
IF NOT EXISTS (SELECT 1 FROM AspNetRoles WHERE Name = 'SysAdmin')
BEGIN
INSERT INTO AspNetRoles
(
Id,
Name,
NormalizedName,
ConcurrencyStamp
)
VALUES
(
NEWID(),
'SysAdmin',
'SYSADMIN',
NEWID()
);
END
GO
IF NOT EXISTS (SELECT 1 FROM AspNetRoles WHERE Name = 'Administrator')
BEGIN
INSERT INTO AspNetRoles
(
Id,
Name,
NormalizedName,
ConcurrencyStamp
)
VALUES
(
NEWID(),
'Administrator',
'ADMINISTRATOR',
NEWID()
);
END
GO
IF NOT EXISTS (SELECT 1 FROM AspNetRoles WHERE Name = 'User')
BEGIN
INSERT INTO AspNetRoles
(
Id,
Name,
NormalizedName,
ConcurrencyStamp
)
VALUES
(
NEWID(),
'User',
'USER',
NEWID()
);
END
GO
/* Verify roles */
SELECT *
FROM AspNetRoles
ORDER BY Name;
GO
